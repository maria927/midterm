plsql.execute <<-SQL
 -- CREACION TABLESPACE SIN PUNTO Y COMA
 -- EJEMPLO CREATE TABLESPACE ...

CREATE TABLESPACE mid_term DATAFILE 'maria.DBF' SIZE 20M ONLINE

SQL

plsql.execute <<-SQL
 -- CREACION PERFIL SIN PUNTO Y COMA
 -- EJEMPLO CREATE PROFILE ...

  CREATE PROFILE exam LIMIT 
    IDLE_TIME 5
    FAILED_LOGIN_ATTEMPTS 2

SQL

plsql.execute <<-SQL
 -- CREACION USUARIO SIN PUNTO Y COMA
 -- EJEMPLO CREATE USER

CREATE USER ejercicios 
	IDENTIFIED BY ejercicios
	DEFAULT TABLESPACE mid_term 
	QUOTA UNLIMITED ON mid_term
	PROFILE exam;

GRANT CONNECT,DBA TO ejercicios


SQL
